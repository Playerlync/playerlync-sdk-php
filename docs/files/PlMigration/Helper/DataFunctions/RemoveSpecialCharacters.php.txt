namespace PlMigration\Helper\DataFunctions;


class RemoveSpecialCharacters implements IValueManipulator
{
    public function execute($value)
    {
        return preg_replace('/[:\*\?"%<>#|\/\\\\]/', '~~', $value);
    }
}
